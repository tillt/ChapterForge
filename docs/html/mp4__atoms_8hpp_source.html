23inline uint32_t fourcc(const char a, const char b, const char c, const char d) {
24 return (uint32_t(uint8_t(a)) << 24) | (uint32_t(uint8_t(b)) << 16) |
25 (uint32_t(uint8_t(c)) << 8) | (uint32_t(uint8_t(d)));
26}
27
28inline uint32_t fourcc(const char t[4]) { return fourcc(t[0], t[1], t[2], t[3]); }
29
30inline uint32_t fourcc(const std::string &s) {
31 if (s.size() < 4)
32 throw std::runtime_error("fourcc string too short");
33 return fourcc(s[0], s[1], s[2], s[3]);
34}
