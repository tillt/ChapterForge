29inline constexpr chapterforge::LogVerbosity ch_severity_for_tag(std::string_view tag) {
30 if (tag == "error") {
31 return chapterforge::LogVerbosity::Error;
32 }
33 if (tag == "warn" || tag == "warning") {
34 return chapterforge::LogVerbosity::Warn;
35 }
36 if (tag == "info") {
37 return chapterforge::LogVerbosity::Info;
38 }
39 // Everything else (io/parser/meta/etc.) treated as debug-level.
40 return chapterforge::LogVerbosity::Debug;
41}

43inline bool ch_should_log(const char* level) {
44 const auto current = chapterforge::get_log_verbosity();
45 const auto sev = ch_severity_for_tag(level ? level : "");
46 return static_cast<int>(sev) <= static_cast<int>(current);
47}
